def A(p, c, tau):
    mu1 = p[0]
    k1 = p[1]
    mu2 = p[2]
    k2 = p[3]
    s1 = c[0]
    s2 = c[1]
    cor = c[2]
    
    Term1 = ((mu1/k1) +(mu2/k2) - (s1**2/(2*k1)) - (s2**2/(2*k2)) - (s1*s2*cor/(k1*k2))) * tau
    Term2 = ((s1**2/k1**3) + (s1*s2*cor/(2*(k1**2)*k2)) - (mu1/k1**2)) * (1-np.exp(-k1*tau))
    Term3 = ((s2**2/k2**3) + (s1*s2*cor/(2*(k2**2)*k2)) - (mu2/k2**2)) * (1-np.exp(-k2*tau))
    Term4 = - (s1**2/(4*k1**3)) * (1-np.exp(-2*k1*tau))
    Term5 = - (s2**2/(4*k2**3)) * (1-np.exp(-2*k2*tau))
    Term6 = - ((s1*s2*cor/(k1*k2*(k1+k2)))) * ((1-np.exp(-(k1 + k2)*tau)))
    A = Term1 + Term2 + Term3 + Term4 + Term5 + Term6
            
    return A

def half_life(df):
    mu = df.mean()
    df_t = df[:-1] - mu
    df_lag = df[1:] - mu
    a = np.matrix(df_t) * np.matrix(df_lag).T
    b = np.matrix(df_t) * np.matrix(df_t).T
    l = -math.log(a/b)
    h = math.log(2)/l
    return h
